    - HomePage.vue                         -->>首页页面

    - Classification.vue                   -->>分类页面

    - ShoppingCart.vue                     -->>购物车页面
 
    - Personal.vue                         -->>个人中心页面

 - details                              -->>详情页，与index同级，其中的页面没有 TabBar组件

    - index.vue                            -->>详情页

    - Search.vue                           -->>搜索页
